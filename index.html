// Aggiungi event listener per aggiornare il filtro e il blocco quando cambia mese/anno const fieldMonthSelect = document.getElementById('field-month'); const fieldYearSelect = document.getElementById('field-year'); const publicMonthSelect = document.getElementById('public-month'); const publicYearSelect = document.getElementById('public-year'); // Event listener per il Servizio di Campo if (fieldMonthSelect) { const originalOnChange = fieldMonthSelect.getAttribute('onchange'); if (originalOnChange) { fieldMonthSelect.setAttribute('onchange', `${originalOnChange}; updateFieldFilterOptions(); updateFieldLockUI();`); } else { fieldMonthSelect.addEventListener('change', function() { updateFieldFilterOptions(); updateFieldLockUI(); renderFieldCalendar(); }); } } if (fieldYearSelect) { const originalOnChange = fieldYearSelect.getAttribute('onchange'); if (originalOnChange) { fieldYearSelect.setAttribute('onchange', `${originalOnChange}; updateFieldFilterOptions(); updateFieldLockUI();`); } else { fieldYearSelect.addEventListener('change', function() { updateFieldFilterOptions(); updateFieldLockUI(); renderFieldCalendar(); }); } } // Event listener per l'Opera Pubblica if (publicMonthSelect) { const originalOnChange = publicMonthSelect.getAttribute('onchange'); if (originalOnChange) { publicMonthSelect.setAttribute('onchange', `${originalOnChange}; updatePublicLockUI();`); } else { publicMonthSelect.addEventListener('change', function() { updatePublicLockUI(); renderPublicCalendar(); }); } } if (publicYearSelect) { const originalOnChange = publicYearSelect.getAttribute('onchange'); if (originalOnChange) { publicYearSelect.setAttribute('onchange', `${originalOnChange}; updatePublicLockUI();`); } else { publicYearSelect.addEventListener('change', function() { updatePublicLockUI(); renderPublicCalendar(); }); } }
